BGP configuration example for IXP Peers

YOUR AS: 11111 IXP Neighbour AS: 22222

router bgp 11111
 template peer-policy ixp-peers //Creat BGP Template for assign to all ixp peers
  route-map ixp-peers-in in
  route-map ixp-peers-out out
  next-hop-self
  soft-reconfiguration inbound
  send-community both
 exit-peer-policy
 !
 neighbor 10.201.x.x remote-as 22222 neighbor 10.201.x.x description IXP-NEIGHBOUR
 neighbor 10.201.x.x  update-source GigabitEthernet4/0/4.262
 neighbor 10.201.x.x  inherit peer-policy ixp-peers         // Assign BGP Template to Peers    


route-map ixp-peers-in deny 100
 match as-path 499
!
route-map ixp-peers-in deny 110
 match ip address prefix-list RFC1918 // Reject RFC1918 , private and multicast IP's from neighbours
!
route-map ixp-peers-in permit 150
 match ip address prefix-list IXP   //Prevent received-route less than /24 subnet
!
route-map ixp-peers-in permit 200
 set local-preference 150           // if peer advertised my route to me, my router prefer local bgp route than received-route
!
route-map ixp-peers-out deny 100
 match as-path 500                  // prevent advertise IXP peer route to IXP Peers, Don't become transit AS
!
route-map ixp-peers-out permit 200
 match ip address prefix-list sup_noexp     
 set community no-export            //for no export my route to their IXP peers
!
!
